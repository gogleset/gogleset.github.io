## 10809번 사칙연산

```jsx
var fs = require('fs');
var input = fs.readFileSync('/dev/stdin').toString().split(' ');
var h = parseInt(input[0]);
var m = parseInt(input[1]);

if(h === 0) {
    (m - 45 < 0) ? console.log(`${24-1} ${(60 + m) - 45}`) : console.log(`${h} ${ m - 45}`);
} else {
    (m - 45 < 0) ? console.log(`${h-1} ${(60 + m) - 45}`) : console.log(`${h} ${ m - 45}`);
}
```
